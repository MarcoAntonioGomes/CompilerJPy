int n=20, i, j, tmp;
int v[n];

for(i=0;i<n;i++) {
	cout << "Informe o " << i+1 << "o elemento: ";
	cin >> v[i];
}

cout << "Vetor antes: ";

for(i=0;i<n;i++) {
	if(i > 0 && i < n-1) {
		cout << ",";
	}
	
	cout << v[i];
}

/* ORDENAÇÃO */
for(i=0;i<n;i++) {
	for(j=i+1;j<n;j++) {
		if(v[j] > v[i]) {
			tmp = v[j];
			v[j] = v[i];
			v[i] = tmp;
		}
	}
}

cout << "Vetor depois: ";

for(i=0;i<n;i++) {
	if(i > 0 && i < n-1) {
		cout << ",";
	}
	
	cout << v[i];
}
